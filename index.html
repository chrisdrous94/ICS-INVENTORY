// ICS Inventory — Google Apps Script backend
// Each inventory type is its own sheet tab.
// Columns per tab: ID | Type | Location | Quantity | Threshold
const HEADERS = ['ID', 'Type', 'Location', 'Quantity', 'Threshold', 'Unit', 'Target', 'Expiry', 'Updated'];
const META_SHEET = '_Meta';

// ── Daily digest settings ──────────────────────────────────────────────
// Leave RECIPIENT_EMAIL blank to send to yourself (the script owner).
// Add more addresses separated by commas, e.g. 'you@school.com, store@school.com'.
const RECIPIENT_EMAIL = '';
const DIGEST_HOUR = 7;                 // Hour of day to send (0-23, script's timezone)
const DIGEST_SUBJECT = 'Inventory reorder list';
const SEND_WHEN_NOTHING_LOW = false;   // true = send an "all stocked" email even with no low items
// ───────────────────────────────────────────────────────────────────────

function getSheet(name) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(name);
  if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(HEADERS); }
  else if (sheet.getLastRow() === 0) { sheet.appendRow(HEADERS); }
  return sheet;
}

function listCategories() {
  return SpreadsheetApp.getActiveSpreadsheet().getSheets()
    .map(function (s) { return s.getName(); })
    .filter(function (n) { return n.charAt(0) !== '_'; });
}

function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function getMetaSheet_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(META_SHEET);
  if (!sh) { sh = ss.insertSheet(META_SHEET); sh.appendRow(['Category', 'Color', 'Icon']); sh.hideSheet(); }
  return sh;
}

function readMeta_() {
  const data = getMetaSheet_().getDataRange().getValues();
  const map = {};
  for (let i = 1; i < data.length; i++) {
    if (!data[i][0]) continue;
    map[data[i][0].toString()] = { color: data[i][1] || '', icon: data[i][2] || '' };
  }
  return map;
}

function writeMeta_(category, color, icon) {
  const sh = getMetaSheet_();
  const data = sh.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0].toString() === category) {
      sh.getRange(i + 1, 2).setValue(color || '');
      sh.getRange(i + 1, 3).setValue(icon || '');
      return;
    }
  }
  sh.appendRow([category, color || '', icon || '']);
}

function renameMeta_(oldName, newName) {
  const sh = getMetaSheet_();
  const data = sh.getDataRange().getValues();
  for (let i = 1; i < data.length; i++)
    if (data[i][0].toString() === oldName) { sh.getRange(i + 1, 1).setValue(newName); return; }
}

function deleteMeta_(name) {
  const sh = getMetaSheet_();
  const data = sh.getDataRange().getValues();
  for (let i = data.length - 1; i >= 1; i--)
    if (data[i][0].toString() === name) sh.deleteRow(i + 1);
}

function rowToItem_(row) {
  let expiry = row[7];
  if (expiry instanceof Date) expiry = Utilities.formatDate(expiry, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  else expiry = expiry ? expiry.toString() : '';
  return {
    id: row[0].toString(), type: row[1], location: row[2],
    quantity: parseInt(row[3]) || 0, threshold: parseInt(row[4]) || 0,
    unit: row[5] ? row[5].toString() : '', target: parseInt(row[6]) || 0,
    expiry: expiry, updated: row[8] ? row[8].toString() : ''
  };
}

function doGet(e) {
  const action = (e.parameter.action || 'get');
  if (action === 'categories') return json(listCategories());
  if (action === 'meta') return json(readMeta_());
  const category = e.parameter.category;
  if (!category) return json([]);
  const data = getSheet(category).getDataRange().getValues();
  const out = [];
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === '' && data[i][1] === '') continue;
    out.push(rowToItem_(data[i]));
  }
  return json(out);
}

function doPost(e) {
  if (!e || !e.postData || !e.postData.contents) return json({ error: 'No data' });
  const p = JSON.parse(e.postData.contents);
  const a = p.action;
  const now = new Date().toISOString();

  if (a === 'addCategory') { getSheet(p.category); if (p.color || p.icon) writeMeta_(p.category, p.color, p.icon); return json({ ok: true, categories: listCategories() }); }
  if (a === 'setMeta') { writeMeta_(p.category, p.color, p.icon); return json({ ok: true }); }
  if (a === 'renameCategory') {
    const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(p.category);
    if (sh) sh.setName(p.newName);
    renameMeta_(p.category, p.newName);
    return json({ ok: true, categories: listCategories() });
  }
  if (a === 'deleteCategory') {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sh = ss.getSheetByName(p.category);
    if (sh && ss.getSheets().length > 1) ss.deleteSheet(sh);
    deleteMeta_(p.category);
    return json({ ok: true, categories: listCategories() });
  }
  if (a === 'move') {
    const from = getSheet(p.category);
    const dest = getSheet(p.toCategory);
    const data = from.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][0].toString() === p.id.toString()) {
        const r = data[i];
        dest.appendRow([r[0], r[1], r[2], r[3], r[4], r[5], r[6], r[7], now]);
        from.deleteRow(i + 1);
        break;
      }
    }
    return json({ ok: true });
  }

  const sheet = getSheet(p.category);
  const data = sheet.getDataRange().getValues();

  if (a === 'add') {
    const it = p.item;
    sheet.appendRow([it.id, it.type, it.location, it.quantity, it.threshold || 0, it.unit || '', it.target || 0, it.expiry || '', now]);
  } else if (a === 'update') {
    for (let i = 1; i < data.length; i++)
      if (data[i][0].toString() === p.id.toString()) {
        sheet.getRange(i + 1, 4).setValue(p.quantity);
        sheet.getRange(i + 1, 9).setValue(now);
        break;
      }
  } else if (a === 'edit') {
    const it = p.item;
    for (let i = 1; i < data.length; i++)
      if (data[i][0].toString() === it.id.toString()) {
        sheet.getRange(i + 1, 2, 1, 1).setValue(it.type);
        sheet.getRange(i + 1, 3, 1, 1).setValue(it.location);
        sheet.getRange(i + 1, 5, 1, 1).setValue(it.threshold || 0);
        sheet.getRange(i + 1, 6, 1, 1).setValue(it.unit || '');
        sheet.getRange(i + 1, 7, 1, 1).setValue(it.target || 0);
        sheet.getRange(i + 1, 8, 1, 1).setValue(it.expiry || '');
        sheet.getRange(i + 1, 9, 1, 1).setValue(now);
        break;
      }
  } else if (a === 'delete') {
    for (let i = 1; i < data.length; i++)
      if (data[i][0].toString() === p.id.toString()) { sheet.deleteRow(i + 1); break; }
  }
  return json({ ok: true });
}

//  DAILY LOW-STOCK EMAIL DIGEST
//  Run setupDailyDigest() once to schedule it. Run sendLowStockDigest()
//  any time to test / send immediately.
// ════════════════════════════════════════════════════════════════════════

// Collect every item at or below its reorder level, grouped by category.
function getLowStock() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const groups = [];
  listCategories().forEach(function (name) {
    const data = ss.getSheetByName(name).getDataRange().getValues();
    const items = [];
    for (let i = 1; i < data.length; i++) {
      const type = data[i][1];
      if (data[i][0] === '' && type === '') continue;
      const qty = parseInt(data[i][3]) || 0;
      const thr = parseInt(data[i][4]) || 0;
      if (thr > 0 && qty <= thr) {
        const unit = data[i][5] ? data[i][5].toString() : '';
        const target = parseInt(data[i][6]) || 0;
        const suggest = target > qty ? (target - qty) : 0;
        items.push({ type: type, location: data[i][2], quantity: qty, threshold: thr, unit: unit, suggest: suggest });
      }
    }
    if (items.length) {
      items.sort(function (a, b) { return (a.quantity - a.threshold) - (b.quantity - b.threshold); });
      groups.push({ category: name, items: items });
    }
  });
  return groups;
}

function recipients_() {
  const set = (RECIPIENT_EMAIL || '').trim();
  if (set) return set;
  return Session.getEffectiveUser().getEmail();
}

function escapeHtml_(s) {
  return String(s == null ? '' : s)
    .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;').replace(/'/g, '&#39;');
}

function buildDigestHtml_(groups, totalCount) {
  const dateStr = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'EEEE d MMMM yyyy');
  let rows = '';
  groups.forEach(function (g) {
    rows +=
      '<tr><td colspan="4" style="padding:18px 16px 6px;font:600 13px/1.3 Arial,sans-serif;' +
      'color:#6750A4;text-transform:uppercase;letter-spacing:.04em;">' + escapeHtml_(g.category) + '</td></tr>';
    g.items.forEach(function (it) {
      rows +=
        '<tr>' +
        '<td style="padding:8px 16px;border-top:1px solid #eee;font:14px Arial,sans-serif;color:#1D192B;">' +
          escapeHtml_(it.type) + '</td>' +
        '<td style="padding:8px 16px;border-top:1px solid #eee;font:14px Arial,sans-serif;color:#666;">' +
          escapeHtml_(it.location) + '</td>' +
        '<td align="center" style="padding:8px 16px;border-top:1px solid #eee;font:700 14px Arial,sans-serif;color:#B3261E;">' +
          it.quantity + '</td>' +
        '<td align="center" style="padding:8px 16px;border-top:1px solid #eee;font:14px Arial,sans-serif;color:#999;">' +
          it.threshold + '</td>' +
        '</tr>';
    });
  });

  const header =
    '<tr><td style="padding:10px 16px;font:600 12px Arial,sans-serif;color:#888;">ITEM</td>' +
    '<td style="padding:10px 16px;font:600 12px Arial,sans-serif;color:#888;">LOCATION</td>' +
    '<td align="center" style="padding:10px 16px;font:600 12px Arial,sans-serif;color:#888;">IN STOCK</td>' +
    '<td align="center" style="padding:10px 16px;font:600 12px Arial,sans-serif;color:#888;">REORDER AT</td></tr>';

  return '' +
    '<div style="background:#f4f2f7;padding:24px 0;">' +
    '<table role="presentation" width="100%" cellpadding="0" cellspacing="0"><tr><td align="center">' +
    '<table role="presentation" width="600" cellpadding="0" cellspacing="0" style="max-width:600px;width:100%;background:#fff;border-radius:16px;overflow:hidden;">' +
      '<tr><td style="background:#6750A4;padding:20px 24px;">' +
        '<div style="font:700 18px Arial,sans-serif;color:#fff;">Inventory reorder list</div>' +
        '<div style="font:13px Arial,sans-serif;color:#EADDFF;margin-top:2px;">' + dateStr + '</div>' +
      '</td></tr>' +
      '<tr><td style="padding:16px 24px;font:15px Arial,sans-serif;color:#1D192B;">' +
        '<b>' + totalCount + '</b> item' + (totalCount === 1 ? '' : 's') + ' at or below the reorder level.' +
      '</td></tr>' +
      '<tr><td style="padding:0 8px 8px;"><table role="presentation" width="100%" cellpadding="0" cellspacing="0">' +
        header + rows +
      '</table></td></tr>' +
      '<tr><td style="padding:16px 24px 24px;font:12px Arial,sans-serif;color:#999;border-top:1px solid #eee;">' +
        'Automated daily summary from ICS Inventory. Open the app to update stock.' +
      '</td></tr>' +
    '</table></td></tr></table></div>';
}

// Main entry point — scans all tabs and emails the digest.
function sendLowStockDigest() {
  const groups = getLowStock();
  const total = groups.reduce(function (n, g) { return n + g.items.length; }, 0);

  if (total === 0 && !SEND_WHEN_NOTHING_LOW) return;

  let html, plain, subject;
  if (total === 0) {
    subject = DIGEST_SUBJECT + ' — all stocked';
    html = '<div style="font:15px Arial,sans-serif;color:#1D192B;padding:24px;">Nothing is below its reorder level today.</div>';
    plain = 'Nothing is below its reorder level today.';
  } else {
    subject = DIGEST_SUBJECT + ' (' + total + ')';
    html = buildDigestHtml_(groups, total);
    plain = groups.map(function (g) {
      return g.category + ':\n' + g.items.map(function (it) {
        return '  - ' + it.type + ' (' + it.location + ') - ' + it.quantity + (it.unit ? ' ' + it.unit : '') + ' left, reorder at ' + it.threshold + (it.suggest ? '  →  order ~' + it.suggest : '');
      }).join('\n');
    }).join('\n\n');
  }

  MailApp.sendEmail({
    to: recipients_(),
    subject: subject,
    body: plain,
    htmlBody: html,
    name: 'ICS Inventory'
  });
}

// Run this ONCE to schedule the daily email. Safe to re-run (clears old copies).
function setupDailyDigest() {
  ScriptApp.getProjectTriggers().forEach(function (tr) {
    if (tr.getHandlerFunction() === 'sendLowStockDigest') ScriptApp.deleteTrigger(tr);
  });
  ScriptApp.newTrigger('sendLowStockDigest')
    .timeBased().everyDays(1).atHour(DIGEST_HOUR).create();
}

// Optional: run to stop the daily email.
function removeDailyDigest() {
  ScriptApp.getProjectTriggers().forEach(function (tr) {
    if (tr.getHandlerFunction() === 'sendLowStockDigest') ScriptApp.deleteTrigger(tr);
  });
}
