// Each inventory type is its own sheet tab.
// Columns per tab: ID | Type | Location | Quantity | Threshold
const HEADERS = ['ID', 'Type', 'Location', 'Quantity', 'Threshold'];

function getSheet(name) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(name);
  if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(HEADERS); }
  else if (sheet.getLastRow() === 0) { sheet.appendRow(HEADERS); }
  return sheet;
}

function listCategories() {
  return SpreadsheetApp.getActiveSpreadsheet().getSheets()
    .map(function (s) { return s.getName(); })
    .filter(function (n) { return n.charAt(0) !== '_'; });
}

function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const action = (e.parameter.action || 'get');
  if (action === 'categories') return json(listCategories());
  const category = e.parameter.category;
  if (!category) return json([]);
  const data = getSheet(category).getDataRange().getValues();
  const out = [];
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === '' && data[i][1] === '') continue;
    out.push({
      id: data[i][0].toString(), type: data[i][1], location: data[i][2],
      quantity: parseInt(data[i][3]) || 0, threshold: parseInt(data[i][4]) || 0
    });
  }
  return json(out);
}

function doPost(e) {
  if (!e || !e.postData || !e.postData.contents) return json({ error: 'No data' });
  const p = JSON.parse(e.postData.contents);
  const a = p.action;

  if (a === 'addCategory') { getSheet(p.category); return json({ ok: true, categories: listCategories() }); }
  if (a === 'renameCategory') {
    const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(p.category);
    if (sh) sh.setName(p.newName);
    return json({ ok: true, categories: listCategories() });
  }
  if (a === 'deleteCategory') {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sh = ss.getSheetByName(p.category);
    if (sh && ss.getSheets().length > 1) ss.deleteSheet(sh);
    return json({ ok: true, categories: listCategories() });
  }

  const sheet = getSheet(p.category);
  const data = sheet.getDataRange().getValues();

  if (a === 'add') {
    const it = p.item;
    sheet.appendRow([it.id, it.type, it.location, it.quantity, it.threshold || 0]);
  } else if (a === 'update') {
    for (let i = 1; i < data.length; i++)
      if (data[i][0].toString() === p.id.toString()) { sheet.getRange(i + 1, 4).setValue(p.quantity); break; }
  } else if (a === 'edit') {
    const it = p.item;
    for (let i = 1; i < data.length; i++)
      if (data[i][0].toString() === it.id.toString()) {
        sheet.getRange(i + 1, 2).setValue(it.type);
        sheet.getRange(i + 1, 3).setValue(it.location);
        sheet.getRange(i + 1, 5).setValue(it.threshold || 0);
        break;
      }
  } else if (a === 'delete') {
    for (let i = 1; i < data.length; i++)
      if (data[i][0].toString() === p.id.toString()) { sheet.deleteRow(i + 1); break; }
  }
  return json({ ok: true });
}
